Undo初期設定（Supabase SQL）
Undoを使うには history に batch_id が必要です（このSQLを1回だけ実行）。 
-- 1) 履歴にID（無ければ）
alter table history
add column if not exists id uuid primary key default gen_random_uuid();

-- 2) Undo用バッチID
alter table history
add column if not exists batch_id uuid;

create index if not exists history_batch_id_idx on history(batch_id);
